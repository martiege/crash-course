\begin{frame}[fragile]
    \frametitle{Løkker - while-løkken}

    While-løkker kjører kun hvis en boolsk verdi er True. Det betyr at vi kan sjekke en condition i løkka. 

\begin{python}
while True:
    print("to infinity!")
print("and beyond?")
\end{python}

\end{frame}

\begin{frame}[fragile]
    \frametitle{Løkker - while-løkken}

    Minner om en for-løkke? 

\begin{python}
i = 0
while i < 10:
    print(i)
    i += 1
\end{python}

\end{frame}
